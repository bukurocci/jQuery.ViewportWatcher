jQuery.ViewportWatcher demo

if window width is narrower than 800px, background color turns to red.

sample

$(function() {
	var watcher = $.ViewportWatcher('widthlessthan800', {
		type: 'width',
		min: 0,
		max: 799
	});
	
	watcher.on('inbound', function(evt, data) {
		$('body').css('background-color', "#ff0000");
	});
	
	watcher.on('outbound', function(evt, data) {
		$('body').css('background-color', "#ffffff");
	});
});
